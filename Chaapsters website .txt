<script>
const apiKey = ""; // Runtime Injection

async function generateFoodVisual(prompt, elementId, btnId) {
const loader = document.getElementById('image-loader');
const placeholder = document.getElementById('hero-placeholder');
const imgElement = document.getElementById(elementId);
const btn = document.getElementById(btnId);

loader.classList.remove('hidden');
if (btn) btn.disabled = true;

const url = `https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey}`;

const payload = {
instances: [{
prompt: `Professional cinematic food photography of ${prompt}. A gourmet spread of tandoori soya chaap and paneer tikka on a rustic dark teal slate platter. Garnished with mint leaves, lemon slices, and red onion rings. Steam rising, glowing charcoal embers in background, soft morning light, 8k resolution, appetizing texture.`
}],
parameters: { sampleCount: 1 }
};

let retryCount = 0;
while (retryCount < 5) {
try {
const response = await fetch(url, {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify(payload)
});

if (!response.ok) throw new Error();

const result = await response.json();
const b64 = result.predictions[0].bytesBase64Encoded;

imgElement.src = `data:image/png;base64,${b64}`;
imgElement.classList.remove('hidden');
placeholder.classList.add('hidden');
loader.classList.add('hidden');
return;

} catch (e) {
retryCount++;
await new Promise(r => setTimeout(r, Math.pow(2, retryCount) * 1000));
}
}
loader.classList.add('hidden');
if (btn) btn.disabled = false;
}

window.onload = () => {
document.getElementById('gen-hero-btn').addEventListener('click', () => {
generateFoodVisual("Malai Soya Chaap and Haryali Paneer Tikka served on a platter", "hero-image", "gen-hero-btn");
});
};
</script>
